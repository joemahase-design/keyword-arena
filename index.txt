function syncToSheets(entry) {
  if (!gsUrl) return;
  const params = new URLSearchParams({
    action:       'addScore',
    session:      entry.session      || entry.cls || '',
    name:         entry.name         || '',
    cls:          entry.cls          || '',
    game:         entry.game         || '',
    totalScore:   entry.totalScore   ?? 0,
    quizScore:    entry.quizScore    ?? '',
    networkScore: entry.networkScore ?? '',
    pythonScore:  entry.pythonScore  ?? '',
    pythonTopics: (entry.pythonTopics||[]).join(', '),
    wordleWon:    entry.wordleWon    ? 'true' : 'false',
    hangmanWon:   entry.hangmanWon   ? 'true' : 'false',
    c4Win:        entry.c4Win        || '',
    detail:       entry.detail       || ''
  });
  const url = gsUrl + '?' + params.toString();

  // Method 1: Image beacon ‚Äî works inside Google Sites iframes, bypasses CSP
  try {
    const img = new Image();
    img.src = url;
  } catch(e) {}

  // Method 2: fetch no-cors ‚Äî works outside Google Sites
  try {
    fetch(url, { method: 'GET', mode: 'no-cors' }).catch(()=>{});
  } catch(e) {}

function wordleSubmit(){
  if(wS.done)return;
  const inp=document.getElementById('wp-input');
  const guess=inp.value.toUpperCase().trim();
  if(guess.length!==wS.word.length){
    setMsg('wp-msg',`Need ${wS.word.length} letters!`,'err');
    const row=document.getElementById('wr-'+wS.curRow);
    row.classList.add('shake');setTimeout(()=>row.classList.remove('shake'),400);return;
  }
  const result=wScore(guess,wS.word);
  wReveal(wS.curRow,guess,result);
  wUpdateKB(guess,result);
  inp.value='';
  if(guess===wS.word){
    wS.done=true;wS.won=true;
    setTimeout(()=>{
      setMsg('wp-msg',`‚úÖ Correct! The word was ${wS.word} ‚Äî solved in ${wS.curRow+1} tries!`,'ok');
      saveScore({name:player.name,cls:player.cls,session:player.session,totalScore:100*(7-wS.curRow-1)||10,quizScore:null,networkScore:null,wordleWon:true,hangmanWon:false,c4Win:null,date:new Date().toLocaleString('en-GB'),game:'Wordle',detail:`Word: ${wS.word}, Guesses: ${wS.curRow+1}`});
    },700);
  } else {
    wS.curRow++;
    if(wS.curRow>=wS.maxRows){wS.done=true;setTimeout(()=>setMsg('wp-msg',`‚ùå Out of guesses! Word was: ${wS.word}`,'err'),700);}
  }
}
function wScore(g,w){
  const res=Array(w.length).fill('absent'),wA=w.split(''),gA=g.split('');
  const uW=Array(w.length).fill(false),uG=Array(g.length).fill(false);
  gA.forEach((c,i)=>{if(c===wA[i]){res[i]='correct';uW[i]=uG[i]=true;}});
  gA.forEach((c,i)=>{if(uG[i])return;for(let j=0;j<wA.length;j++){if(!uW[j]&&wA[j]===c){res[i]='present';uW[j]=uG[i]=true;break;}}});
  return res;
}
function wReveal(ri,g,result){
  result.forEach((r,i)=>setTimeout(()=>{
    const cell=document.getElementById(`wc-${ri}-${i}`);
    cell.textContent=g[i];cell.classList.add('flip');
    setTimeout(()=>{cell.classList.remove('flip','filled');cell.classList.add(r);},200);
  },i*100));
}
function wUpdateKB(g,result){
  const pri={correct:3,present:2,absent:1};
  g.split('').forEach((c,i)=>{
    const b=document.querySelector(`[data-key="${c}"]`);if(!b)return;
    const cur=pri[['correct','present','absent'].find(x=>b.classList.contains(x))]||0;
    if((pri[result[i]]||0)>cur){b.classList.remove('correct','present','absent');b.classList.add(result[i]);}
  });
}

function shuffleArray(arr){
  const a=[...arr];
  for(let i=a.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[a[i],a[j]]=[a[j],a[i]];}
  return a;
}

  p1.className='c4-plabel p1'+(c4S.turn===1?' active':'');
  p2.className='c4-plabel p2'+(c4S.turn===2?' active':'');
  p1.innerHTML=`üîµ PLAYER 1 <span id="c4-s1">${c4S.scores[1]}</span>`;
  p2.innerHTML=`PLAYER 2 <span id="c4-s2">${c4S.scores[2]}</span> üüÝ`;
  document.getElementById('c4-turn').textContent=c4S.done?'':
    (c4S.turn===1?'PLAYER 1 üîµ':'PLAYER 2 üüÝ')+' ‚Äî click a column to answer a question!';
}
function c4ColClick(col){
  if(c4S.done||c4S.pendingCol>=0)return;
  if(c4S.board[0][col]!==0){setMsg('c4-msg','‚öÝ Column full!','err');return;}
  c4S.pendingCol=col;
  const q=c4S.shuffledQ[c4S.qIdx++%c4S.shuffledQ.length];
  c4S._curQ=q;
  document.getElementById('c4-col-num').textContent=col+1;
  document.getElementById('c4-qtext').textContent=q.q;
  const opts=document.getElementById('c4-opts');opts.innerHTML='';
  q.opts.forEach((o,i)=>{
    const b=document.createElement('button');b.className='c4-opt';
    b.textContent=String.fromCharCode(65+i)+'. '+o;
    b.onclick=()=>c4Answer(i,q.ans);opts.appendChild(b);});
  document.getElementById('c4-qmsg').style.display='none';
  document.getElementById('c4-qpanel').style.display='block';
}
function c4Answer(idx,ans){
  document.querySelectorAll('.c4-opt').forEach(b=>b.disabled=true);
  const msg=document.getElementById('c4-qmsg');msg.style.display='block';
  if(idx===ans){
    document.querySelectorAll('.c4-opt')[ans].classList.add('correct');
    setMsg('c4-qmsg','‚úÖ Correct! Placing piece‚Ä¶','ok');
    setTimeout(()=>{placeC4();document.getElementById('c4-qpanel').style.display='none';c4S.pendingCol=-1;},900);
  } else {
    document.querySelectorAll('.c4-opt')[ans].classList.add('correct');
    document.querySelectorAll('.c4-opt')[idx].classList.add('wrong');
    setMsg('c4-qmsg','‚ùå Wrong! Turn lost. Answer: '+c4S._curQ.opts[ans],'err');
    setTimeout(()=>{document.getElementById('c4-qpanel').style.display='none';c4S.pendingCol=-1;c4S.turn=c4S.turn===1?2:1;updateC4UI();},1800);
  }
}
function placeC4(){
  const col=c4S.pendingCol;let row=-1;
  for(let r=5;r>=0;r--){if(c4S.board[r][col]===0){row=r;break;}}
  if(row===-1)return;
  c4S.board[row][col]=c4S.turn;
  document.getElementById(`c4-${row}-${col}`).classList.add(c4S.turn===1?'p1':'p2');
  const winner=checkC4Win();
  if(winner){
    c4S.done=true;c4S.scores[winner]++;updateC4UI();
    setMsg('c4-msg',`üèÜ PLAYER ${winner} WINS! (P1:${c4S.scores[1]} P2:${c4S.scores[2]})`,'ok');
    document.querySelectorAll('.c4-cbtn').forEach(b=>b.disabled=true);
    saveScore({name:player.name,cls:player.cls,session:player.session,totalScore:50,quizScore:null,networkScore:null,wordleWon:false,hangmanWon:false,c4Win:`P${winner}`,date:new Date().toLocaleString('en-GB'),game:'Connect4',detail:`Player ${winner} won`});
    return;
  }
  if(c4S.board[0].every(v=>v!==0)){c4S.done=true;setMsg('c4-msg','ü§ù Draw!','info');return;}
  c4S.turn=c4S.turn===1?2:1;updateC4UI();
}
function checkC4Win(){
  const b=c4S.board,dirs=[[0,1],[1,0],[1,1],[1,-1]];
  for(let r=0;r<6;r++)for(let c=0;c<7;c++){
    if(!b[r][c])continue;const p=b[r][c];
    for(const[dr,dc]of dirs){
      let cells=[[r,c]],nr=r+dr,nc=c+dc;
      while(nr>=0&&nr<6&&nc>=0&&nc<7&&b[nr][nc]===p){cells.push([nr,nc]);nr+=dr;nc+=dc;}
      if(cells.length>=4){cells.forEach(([rr,cc])=>document.getElementById(`c4-${rr}-${cc}`).classList.add('win'));return p;}
    }
  }
  return null;
}

const HM_PARTS=['hm-head','hm-body','hm-larm','hm-rarm','hm-lleg','hm-rleg'];
let hmS={};
function initHangman(){
  const kw=shuffleArray(KEYWORDS.filter(k=>k.word.length>=4))[0];
  hmS={word:kw.word.toUpperCase(),hint:kw.hint,cat:kw.category,lives:6,guessed:new Set(),done:false,won:false};
  HM_PARTS.forEach(id=>document.getElementById(id).classList.remove('show'));
  document.getElementById('hm-cat').textContent='[ '+kw.category+' ]';
  document.getElementById('hm-hint').textContent='üí° '+kw.hint;
  document.getElementById('hm-msg').style.display='none';
  renderHMWord();renderHMLives();renderHMGuessed();buildHMKB();
}
function renderHMWord(){
  const d=document.getElementById('hm-word');d.innerHTML='';
  hmS.word.split('').forEach(c=>{
    const el=document.createElement('div');
    el.className=c===' '?'hm-l sp':'hm-l';
    el.textContent=c===' '?'':hmS.guessed.has(c)?c:'';
    d.appendChild(el);});
}
function renderHMLives(){document.getElementById('hm-lives').innerHTML=`LIVES: ${'‚ù§Ô∏è'.repeat(hmS.lives)}${'üñ§'.repeat(6-hmS.lives)}`;}
function renderHMGuessed(){
  const wrong=[...hmS.guessed].filter(c=>!hmS.word.includes(c));
  document.getElementById('hm-guessed').textContent=wrong.length?'Wrong: '+wrong.join(' '):'';
}
function buildHMKB(){
  const kb=document.getElementById('hm-keyboard');kb.innerHTML='';
  'ABCDEFGHIJKLMNOPQRSTUVWXYZ'.split('').forEach(c=>{
    const b=document.createElement('button');b.className='hm-key';b.textContent=c;b.id='hmk-'+c;
    b.onclick=()=>hmGuess(c);kb.appendChild(b);});
}
function hmGuess(c){
  if(hmS.done||hmS.guessed.has(c))return;
  hmS.guessed.add(c);
  const btn=document.getElementById('hmk-'+c);
  if(hmS.word.includes(c)){btn.classList.add('hit');}
  else{btn.classList.add('miss');btn.disabled=true;
    hmS.lives--;document.getElementById(HM_PARTS[6-hmS.lives-1]).classList.add('show');}
  renderHMWord();renderHMLives();renderHMGuessed();
  const won=hmS.word.split('').every(ch=>ch===' '||hmS.guessed.has(ch));
  if(won){
    hmS.done=true;hmS.won=true;
    const msg=document.getElementById('hm-msg');msg.style.display='block';msg.className='msg ok';
    msg.textContent=`‚úÖ You got it! "${hmS.word}"`;
    document.querySelectorAll('.hm-key').forEach(b=>b.disabled=true);
    saveScore({name:player.name,cls:player.cls,session:player.session,totalScore:hmS.lives*10+20,quizScore:null,networkScore:null,wordleWon:false,hangmanWon:true,c4Win:null,date:new Date().toLocaleString('en-GB'),game:'Hangman',detail:`Word: ${hmS.word}, Lives left: ${hmS.lives}`});
    return;
  }
  if(hmS.lives<=0){
    hmS.done=true;
    hmS.word.split('').forEach(c=>{if(c!==' ')hmS.guessed.add(c);});
    renderHMWord();
    const msg=document.getElementById('hm-msg');msg.style.display='block';msg.className='msg err';
    msg.textContent=`‚ùå Out of lives! Word was "${hmS.word}".`;
    document.querySelectorAll('.hm-key').forEach(b=>b.disabled=true);
  }
}

let ncS={};
function startNC(){
  ncS={qs:shuffleArray(NC_Q).slice(0,20),cur:0,score:0,streak:0,maxStreak:0,
       correct:0,results:[],timer:null,timeLeft:20,startTime:Date.now(),answered:false,qStart:0};
  document.getElementById('nc-splash').style.display='none';
  document.getElementById('nc-results').style.display='none';
  document.getElementById('nc-game').style.display='block';
  document.getElementById('nc-hud-player').innerHTML=player.name+'<small>'+player.cls+'</small>';
  ncBeginQ();
}
function ncBeginQ(){
  if(ncS.cur>=ncS.qs.length){ncEnd();return;}
  const q=ncS.qs[ncS.cur];
  ncS.answered=false;ncS.qStart=Date.now();
  document.getElementById('nc-qnum').textContent=`Q ${ncS.cur+1}/20`;
  document.getElementById('nc-score').textContent=ncS.score;
  const strk=document.getElementById('nc-streak');
  strk.innerHTML=`STREAK<br><span>${ncS.streak}${ncS.streak>=3?'üî•':''}</span>`;
  // dots
  const dots=document.getElementById('nc-dots');dots.innerHTML='';
  ncS.qs.forEach((_,i)=>{
    const d=document.createElement('div');d.className='nc-dot';
    if(i<ncS.cur)d.classList.add(ncS.results[i]?.correct?'c':'w');
    if(i===ncS.cur)d.classList.add('cur');
    dots.appendChild(d);});
  document.getElementById('nc-qcat').textContent='[ '+q.cat+' ]';
  document.getElementById('nc-qtext').textContent=q.q;
  const opts=document.getElementById('nc-opts');opts.innerHTML='';
  const shuffled=q.opts.map((o,i)=>({o,i})).sort(()=>Math.random()-.5);
  const correctText=q.opts[q.ans];
  shuffled.forEach(({o},idx)=>{
    const b=document.createElement('button');b.className='nc-opt';
    b.innerHTML=`<span class="ol">${String.fromCharCode(65+idx)}.</span>${o}`;
    b.onclick=()=>ncAnswer(o===correctText,b,q);
    opts.appendChild(b);});
  const fb=document.getElementById('nc-fb');fb.className='nc-fb';fb.style.display='none';
  ncStartTimer();
}
function ncStartTimer(){
  ncS.timeLeft=20;clearInterval(ncS.timer);
  const bar=document.getElementById('nc-tbar');
  ncS.timer=setInterval(()=>{
    ncS.timeLeft-=.1;
    bar.style.width=Math.max(0,(ncS.timeLeft/20)*100)+'%';
    if(ncS.timeLeft<=0){clearInterval(ncS.timer);ncTimeout();}
  },100);
}
function ncTimeout(){
  if(ncS.answered)return;ncS.answered=true;
  const q=ncS.qs[ncS.cur];ncS.streak=0;
  ncS.results.push({q:q.q,cat:q.cat,correct:false,points:0,timeTaken:20});
  document.querySelectorAll('.nc-opt').forEach(b=>{if(b.textContent.includes(q.opts[q.ans]))b.classList.add('correct');b.disabled=true;});
  const fb=document.getElementById('nc-fb');
  fb.textContent=`‚è∞ Time's up! Answer: ${q.opts[q.ans]}`;
  fb.className='nc-fb bad show';fb.style.display='block';
  setTimeout(()=>{ncS.cur++;ncBeginQ();},2000);
}
function ncAnswer(isCorrect,clickedBtn,q){
  if(ncS.answered)return;ncS.answered=true;
  clearInterval(ncS.timer);
  const timeTaken=parseFloat(((Date.now()-ncS.qStart)/1000).toFixed(1));
  let pts=0;
  const fb=document.getElementById('nc-fb');
  document.querySelectorAll('.nc-opt').forEach(b=>b.disabled=true);
  if(isCorrect){
    const tBonus=Math.max(20,Math.round((ncS.timeLeft/20)*100));
    const sBonus=ncS.streak>=3?Math.min(ncS.streak*5,50):0;
    pts=tBonus+sBonus;ncS.score+=pts;ncS.streak++;ncS.maxStreak=Math.max(ncS.maxStreak,ncS.streak);ncS.correct++;
    clickedBtn.classList.add('correct');
    fb.textContent=`‚úÖ Correct! +${pts} pts${sBonus?' (üî• streak bonus)':''}`;
    fb.className='nc-fb ok show';fb.style.display='block';
    const pf=document.getElementById('pts-flash');
    pf.textContent='+'+pts;pf.style.color=ncS.streak>=3?'var(--accent2)':'var(--green)';
    pf.className='pts-flash show';setTimeout(()=>pf.className='pts-flash',800);
  } else {
    ncS.streak=0;clickedBtn.classList.add('wrong');
    document.querySelectorAll('.nc-opt').forEach(b=>{if(b.textContent.includes(q.opts[q.ans]))b.classList.add('correct');});
    fb.textContent=`‚ùå Incorrect. Answer: ${q.opts[q.ans]}`;
    fb.className='nc-fb bad show';fb.style.display='block';
  }
  document.getElementById('nc-score').textContent=ncS.score;
  document.getElementById('nc-streak').innerHTML=`STREAK<br><span>${ncS.streak}${ncS.streak>=3?'üî•':''}</span>`;
  ncS.results.push({q:q.q,cat:q.cat,correct:isCorrect,points:pts,timeTaken});
  setTimeout(()=>{ncS.cur++;ncBeginQ();},isCorrect?1300:2000);
}
function ncEnd(){
  clearInterval(ncS.timer);
  const totalTime=parseFloat(((Date.now()-ncS.startTime)/1000).toFixed(1));
  document.getElementById('nc-game').style.display='none';
  document.getElementById('nc-results').style.display='block';

  document.getElementById('nc-r-correct').textContent=ncS.correct+'/20';
  const acc=Math.round((ncS.correct/20)*100);

  document.getElementById('nc-r-time').textContent=totalTime+'s';
  const gb=document.getElementById('nc-grade');

  const [,label,col]=grades.find(([t])=>acc>=t);
  gb.textContent=label;gb.style.borderColor=col;gb.style.color=col;
  const bl=document.getElementById('nc-breakdown');bl.innerHTML='';

  saveScore({name:player.name,cls:player.cls,session:player.session,totalScore:ncS.score,quizScore:null,networkScore:ncS.score,wordleWon:false,hangmanWon:false,c4Win:null,date:new Date().toLocaleString('en-GB'),game:'NetworkChallenge',detail:`Score:${ncS.score}, Acc:${acc}%, Time:${totalTime}s`});
}

function pyAnswer(btn){

  pyState.quizAnswered = true;
  const chosen = btn.dataset.answer;
  const correct = btn.dataset.correct;
  const t = pyState.current;
  const q = t.quiz[pyState.quizIdx];
  document.querySelectorAll('.py-qopt').forEach(b=>{
    b.disabled=true;
    if(b.dataset.answer===correct) b.classList.add('correct');
  });
  if(chosen===correct){
    btn.classList.add('correct');
    pyState.quizScore++;
  } else {
    btn.classList.add('wrong');
  }
  const expEl = document.getElementById('py-exp');
  expEl.textContent = 'üí° '+q.exp;
  expEl.style.display = 'block';
  document.getElementById('py-next-btn').style.display = 'inline-block';
}

    setTimeout(()=>btn.textContent='COPY',1500);
  });
}

// ‚îÄ‚îÄ DATA ‚îÄ‚îÄ
async function tLoadData(){
  let all = [];

  // Fetch from Google Sheets - primary cross-device source
  if(gsUrl){
    try {
      const res = await fetch(gsUrl + '?action=getScores&t=' + Date.now(), {
        method: 'GET',
        cache: 'no-store'
      });
      const txt = await res.text();
      const data = JSON.parse(txt);
      if(data.status === 'ok' && Array.isArray(data.scores)){
        all = data.scores;
        tSetGsStatus('sheets');
      }
    } catch(e){ /* fall through to in-memory */ }
  }

  // Also add any in-memory scores from this device not yet in Sheets
  const inMem = (typeof leaderboard !== 'undefined') ? leaderboard : [];
  inMem.forEach(entry => {
    if(!all.some(s => s.name===entry.name && s.date===entry.date && s.game===entry.game)){
      all.push(entry);
    }
  });

  // Only fall back to localStorage if Sheets fetch completely failed (not just empty)
  // An empty Sheets response means it was cleared - don't re-add old local scores
  if(!all.length && !gsUrl){
    const currentKey = store.get('cba_session_key');
    const stored = JSON.parse(store.get('cba_arena_lb') || '[]');
    all = currentKey ? stored.filter(e => e._sessionKey === currentKey) : stored;
  }

  tLeaderboard = tAggregateScores(all);
  tLeaderboard.sort((a,b) => b.totalScore - a.totalScore);
  tRenderLB(); tRenderStats(); tRenderPyProgress();

  const rp = document.getElementById('t-results-panel');
  if(rp && rp.style.display !== 'none') tRenderResultsTable();

  const luEl = document.getElementById('last-update');
  if(luEl) luEl.textContent = 'Last update: ' + new Date().toLocaleTimeString('en-GB');
}

// Aggregate raw score rows into one entry per student (best scores)
function tAggregateScores(rows){
  const map = {};
  rows.forEach(e=>{
    const key = e.name+'||'+e.cls;
    if(!map[key]){
      map[key] = {
        name:e.name, cls:e.cls,
        totalScore:0, quizScore:null, networkScore:null,
        pythonScore:null, pythonTopics:[],
        wordleWon:false, hangmanWon:false, c4Win:null,
        game:e.game, date:e.date
      };
    }
    const s = map[key];
    s.totalScore += (e.totalScore||0);
    if(e.quizScore!=null)    s.quizScore    = Math.max(s.quizScore||0, e.quizScore);
    if(e.networkScore!=null) s.networkScore = Math.max(s.networkScore||0, e.networkScore);
    if(e.pythonScore!=null)  s.pythonScore  = (s.pythonScore||0) + e.pythonScore;
    if(e.pythonTopics)       s.pythonTopics = [...new Set([...s.pythonTopics,...e.pythonTopics])];
    if(e.wordleWon)  s.wordleWon  = true;
    if(e.hangmanWon) s.hangmanWon = true;
    if(e.c4Win)      s.c4Win      = e.c4Win;
    s.game = e.game; s.date = e.date;
  });
  return Object.values(map);
}

// ‚îÄ‚îÄ RESULTS TABLE ‚Äî shows every individual score entry ‚îÄ‚îÄ
async function tRenderResultsTable() {
  // Fetch latest from Sheets first if available
  if(gsUrl){
    try {
      const res  = await fetch(gsUrl + '?action=getScores&t=' + Date.now());
      const data = await res.json();
      if(data.status === 'ok' && Array.isArray(data.scores) && data.scores.length){
        window._sheetsRaw = data.scores;
      }
    } catch(e) {}
  }
  const tbody = document.getElementById('t-results-body');
  const countEl = document.getElementById('t-res-count');
  if (!tbody) return;

  // Combine Sheets + in-memory
  const sheetsRaw = window._sheetsRaw || [];
  const inMem = (typeof leaderboard !== 'undefined') ? leaderboard : [];
  const raw = [...sheetsRaw];
  inMem.forEach(e => {
    if(!raw.some(s => s.name===e.name && s.date===e.date && s.game===e.game)) raw.push(e);
  });

  if (!raw.length) {
    tbody.innerHTML = '<tr><td colspan="11" style="padding:2rem;color:var(--dim);font-family:Share Tech Mono,monospace;font-size:.75rem;text-align:center;">No results yet ‚Äî waiting for students to play</td></tr>';
    if(countEl) countEl.textContent = 'No results yet';
    return;
  }

  if(countEl) countEl.textContent = raw.length + ' result' + (raw.length!==1?'s':'') + ' ¬∑ ' + [...new Set(raw.map(e=>e.name))].length + ' students';

  tbody.innerHTML = [...raw].reverse().map(e => {
    const score = e.totalScore || 0;
    const scoreColor = score >= 1000 ? 'var(--green)' : score >= 500 ? 'var(--yellow)' : 'var(--accent)';
    return `<tr>
      <td class="td-name">${e.name||'‚Äî'}</td>
      <td class="td-cls">${e.cls||'‚Äî'}</td>
      <td class="td-game">${(e.game||'‚Äî').replace('NetworkChallenge','Network').replace('Connect4','Connect 4')}</td>
      <td class="td-score" style="color:${scoreColor}">${score}</td>
      <td>${e.quizScore!=null ? e.quizScore+'/10' : '<span class="td-no">‚Äî</span>'}</td>
      <td>${e.networkScore!=null ? e.networkScore : '<span class="td-no">‚Äî</span>'}</td>
      <td>${e.pythonScore!=null ? e.pythonScore+'pts' : '<span class="td-no">‚Äî</span>'}</td>
      <td class="${e.wordleWon?'td-yes':'td-no'}">${e.wordleWon?'‚úì':'‚Äî'}</td>
      <td class="${e.hangmanWon?'td-yes':'td-no'}">${e.hangmanWon?'‚úì':'‚Äî'}</td>
      <td class="${e.c4Win?'td-yes':'td-no'}">${e.c4Win||'‚Äî'}</td>
      <td class="td-game">${e.date||'‚Äî'}</td>
    </tr>`;
  }).join('');
}
